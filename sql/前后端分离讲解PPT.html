// 401：Token 过期处理
if (status === 401 && !_isRetry) {
  if (isRefreshing) {
    // 加入等待队列
    return new Promise((resolve) => {
      requestsQueue.push((token) => {
        originalRequest.headers
          .Authorization = `Bearer ${token}`
        resolve(service(originalRequest))
      })
    })
  }
  isRefreshing = true
  // 调用刷新接口获取新 Token

  processQueue(res.data.token)
  // 重试原请求
  return service(originalRequest)
}

Setup Store 语法（Composition API）
export const useAuthStore = defineStore('auth', () => {
  // ===== 响应式状态（类似 data） =====
  const token = ref<string | null>(getToken())
  const isLoggedIn = ref(isAuthenticated())
  const isAdminUser = ref(checkIsAdmin())
  const currentUsername = ref('')

  // ===== 计算属性 =====
  const authHeader = computed(() =>
    token.value ? `Bearer ${token.value}` : null
  )

  // ===== 方法 =====
  async function login(params) { ... }
  async function logout() { ... }
  function hasPerm(key) { ... }

  return { token, isLoggedIn, isAdminUser,
           authHeader, login, logout, hasPerm }
})

🖥️ Spring Boot Controller — RESTful API 设计

AuthController — 认证接口 @RestController 
@RestController
@RequestMapping("/api/auth")
public class AuthController {

  @Autowired private AuthService authService;

  // 登录 → 返回 JWT Token
  @PostMapping("/login")
  public Map<String, Object> login(
    @RequestBody Map<String, String> body
  ) {
    return authService.login(
      body.get("username"),
      body.get("password")
    );
  }

  // 登出 → 清除 Redis 在线状态
  @PostMapping("/logout")
  public Map<String, Object> logout() {
    return authService.logout();
  }
}
@RestController = @Controller + @ResponseBody 直接返回 Map（不用 ResultDto）
分层架构 Controller → Service → Mapper

🌐 CORS 跨域 — 前后端分离的关键配置

为什么需要 CORS？
浏览器同源策略要求协议 + 域名 + 端口完全一致。前端跑在 localhost:5173，后端在 localhost:8082，属于不同源，浏览器会拦截请求。

// Spring Boot CORS 配置
@Configuration
public class CorsConfig implements WebMvcConfigurer {
  @Override
  public void addCorsMappings(CorsRegistry r) {
    r.addMapping("/api/**")
     .allowedOrigins("http://localhost:5173")
     .allowedMethods("GET","POST","PUT","DELETE")
     .allowedHeaders("*")
     .allowCredentials(true);
  }
}
